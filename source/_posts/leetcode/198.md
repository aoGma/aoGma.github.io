## [打家劫舍](https://leetcode.cn/problems/house-robber/)

一开始以为只有两个答案二选一，大意了

```js
/**
 * @param {number[]} nums
 * @return {number}
 */
var rob = function (nums) {
 let oddSum = 0,
  evenSum = 0;
 for (let i = 0; i < nums.length; i++) {
  if (i % 2 === 0) {
   evenSum += nums[i];
  } else {
   oddSum += nums[i];
  }
 }
 return Math.max(oddSum, evenSum);
};
```
