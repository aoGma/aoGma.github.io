/**
 * @param {character[]} s
 * @return {void} Do not return anything, modify s in-place instead.

var reverseString = function (s) {
 const len = s.length;
 const mid = Math.floor(s.length / 2);
 for (let i = 0; i < mid; i++) {
  const temp = s[i];
  s[i] = s[len - i - 1];
  s[len - i - 1] = temp;
 }
 // console.log("ğŸš€ ~ reverseString ~ s:", s);
 return s;
};
