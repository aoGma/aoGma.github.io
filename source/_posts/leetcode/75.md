var sortColors = function (nums) {
  // å¿«é€Ÿæ’åº
  if (nums.length <= 1) return nums;
  const pivot = nums[0];
  const left = nums.slice(1).filter((x) => x < pivot);
  const right = nums.slice(1).filter((x) => x >= pivot);
  return [...sortColors(left), pivot, ...sortColors(right)];
};
